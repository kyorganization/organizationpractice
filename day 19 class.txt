# ğŸ” Regex Examples
import re

# [^arn] â†’ match any character except a, r, n
txt = "the rain in spain"
print(re.findall("[^arn]", txt))

# [0123] â†’ match digits 0,1,2,3
txt = "the rain 1 in 4 3 spain"
print(re.findall("[0123]", txt))

# [0-9] â†’ match all digits
txt = "the rain1 in 4 3 spain"
print(re.findall("[0-9]", txt))

# [0-5][0-9] â†’ match minutes (00â€“59)
txt = "the time is 11:45 AM"
print(re.findall("[0-5][0-9]", txt))

# [a-zA-Z] â†’ match all letters
txt = "the rain1 in 4 3 spain"
print(re.findall("[a-zA-Z]", txt))

# [+] â†’ matches literal plus sign
txt = "the rain in spain"
x = re.search(r"\s", txt)   # \s â†’ whitespace
print(x)
print(x.start(), x.end())

# Search for "spain"
txt = "the rain in spain"
x = re.search("spain", txt)
print(x)
print(x.start(), x.end(), x.span(), x.group())

# split() â†’ split string at matches
txt = "the rain in and moon in sky"
print(re.split("in", txt))

# split with \s â†’ whitespace
txt = "the rain in and moon in sky"
print(re.split(r"\s", txt, 1))   # split once

# sub() â†’ replace text
txt = "the rain in and moon in sky"
print(re.sub(r"\s", "8", txt))

## âœ… Why logging?
