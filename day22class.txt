# ğŸ“… Date Class â†’ working only with dates (year, month, day)

import datetime

today = datetime.date.today()
print("Today's date:", today)

# YY-MM-DD format
d = datetime.date(2025, 12, 10)
print("Custom date:", d)
print("Type:", type(d))

print("Year:", d.year)
print("Month:", d.month)
print("Day:", d.day)

# ğŸ¯ Useful attributes of date object
# - year
# - month
# - day

# weekday() â†’ Monday=0 ... Sunday=6
# isoweekday() â†’ Monday=1 ... Sunday=7

d = datetime.date(2025, 12, 11)
print("weekday():", d.weekday())       # 3 â†’ Thursday
print("isoweekday():", d.isoweekday()) # 4 â†’ Thursday


# â° TIME Class â†’ working only with time (hours, minutes, seconds, microseconds)

t = datetime.time(hour=12, minute=30, second=15)
print("Time object:", t)
print("Type:", type(t))

# Attributes
print("Hour:", t.hour)
print("Minute:", t.minute)
print("Second:", t.second)
print("Microsecond:", t.microsecond)


# ğŸ“† DATETIME Class â†’ working with both date and time

dt = datetime.datetime.now()
print("Current datetime:", dt)

dt = datetime.datetime(2025, 12, 10, 12, 30, 20)
print("Custom datetime:", dt)

# Arguments order:
# year, month, day, hour, minute, second, microsecond


# ğŸ¨ Formatting Dates and Times

now = datetime.datetime.now()
print("Now:", now)
print("Type:", type(now))

# strftime â†’ format datetime into string
formatted = now.strftime("%d-%m-%Y %H:%M:%S")
print("Formatted:", formatted)

# ğŸ“Š Common Directives
# %a â†’ Weekday short (Wed)
# %A â†’ Weekday full (Wednesday)
# %d â†’ Day of month (01â€“31)
# %b â†’ Month short (Dec)
# %B â†’ Month full (December)
# %m â†’ Month number (01â€“12)
# %y â†’ Year short (25)
# %Y â†’ Year full (2025)
# %H â†’ Hour (00â€“23)
# %I â†’ Hour (00â€“12)
# %p â†’ AM/PM
# %M â†’ Minute (00â€“59)
# %S â†’ Second (00â€“59)
# %f â†’ Microsecond
# %Z â†’ Timezone
# %U â†’ Week number (Sunday first)
# %W â†’ Week number (Monday first)

x = datetime.datetime.now()
print("Today's date is:", x)
print("Month + Weekday:", x.strftime("%m-%A"))
b = x.strftime("%B")
print("Month name:", b)
print("Type:", type(b))


# ğŸ“¥ Parsing Strings into datetime
s = "10-12-2025 14:30:00"
dt = datetime.datetime.strptime(s, "%d-%m-%Y %H:%M:%S")
print("Parsed datetime:", dt)

formatted = dt.strftime("%d-%m-%Y-%A %H:%M:%S")
print("Formatted again:", formatted)


# ğŸ”„ Time Differences (timedelta)

d1 = datetime.date(2025, 12, 10)
d2 = datetime.date(2025, 12, 25)
diff = d2 - d1
print("Difference:", diff)
print("Type:", type(diff))
print("Start date:", d1)


# ğŸ‚ Age Calculation Example

dob = datetime.date(2001, 1, 31)
today = datetime.date.today()

age = today.year - dob.year
if (today.month, today.day) < (dob.month, dob.day):
    age -= 1

print("Age:", age)
